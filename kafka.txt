services:
  # To jest Twój broker (zamiennik Kafki)
  redpanda:
    image: docker.redpanda.com/redpandadata/redpanda:latest
    container_name: redpanda
    command:
      - redpanda start
      - --smp 1  # Użyj tylko 1 rdzenia (oszczędność zasobów)
      - --memory 1G # Limit pamięci
      - --reserve-memory 0M
      - --overprovisioned
      - --node-id 0
      - --check=false
      # Konfiguracja sieci: Port 9092 dla Twojej aplikacji .NET (localhost), 29092 dla UI wewnątrz dockera
      - --kafka-addr PLAINTEXT://0.0.0.0:29092,OUTSIDE://0.0.0.0:9092
      - --advertise-kafka-addr PLAINTEXT://redpanda:29092,OUTSIDE://localhost:9092
      - --pandaproxy-addr 0.0.0.0:8082
      - --advertise-pandaproxy-addr localhost:8082
    ports:
      - 9092:9092 # Port dla Twojej aplikacji .NET
      - 8081:8081 # Schema Registry (opcjonalne, ale przydatne)
    volumes:
      - redpanda-data:/var/lib/redpanda/data

  # To jest Twoje UI (Interfejs graficzny)
  console:
    image: docker.redpanda.com/redpandadata/console:latest
    container_name: redpanda-console
    ports:
      - 8080:8080 # Tutaj wchodzisz przez przeglądarkę
    environment:
      # UI musi wiedzieć, gdzie jest Kafka (wewnątrz sieci dockera to port 29092)
      KAFKA_BROKERS: redpanda:29092
    depends_on:
      - redpanda

volumes:
  redpanda-data:
